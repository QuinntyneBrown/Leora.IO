using System;
using System.Collections.Generic;
using {{ name }}.Dtos;
using {{ name }}.Data;
using {{ name }}.Models;
using System.Linq;

namespace {{ name }}.Services
{
    public class Service : IService
    {
        public Service(IUow uow, ICacheProvider cacheProvider)
        {
            _uow = uow;
            _repository = uow.{{ entity }}s;
            _cache = cacheProvider.GetCache();
        }

        public {{ entity }}AddOrUpdateResponseDto AddOrUpdate({{ entity }}AddOrUpdateRequestDto request)
        {
            var entity = _repository.GetAll()
                .FirstOrDefault(x => x.Id == request.Id && x.IsDeleted == false);
            if (entity == null) _repository.Add(entity = new {{ entity }}());
            entity.Name = request.Name;
            _uow.SaveChanges();
            return new {{ entity }}AddOrUpdateResponseDto(entity);
        }

        public ICollection<{{ entity }}Dto> Get()
        {
            ICollection<{{ entity }}Dto> response = new HashSet<{{ entity }}Dto>();
            var entities = _repository.GetAll().Where(x => x.IsDeleted == false).ToList();
            foreach (var entity in entities) { response.Add(new {{ entity }}Dto(entity)); }
            return response;
        }

        public {{ entity }}Dto GetById(int id)
        {
            return new {{ entity }}Dto(_repository.GetAll().Where(x => x.Id == id && x.IsDeleted == false).FirstOrDefault());
        }

        public dynamic Remove(int id)
        {
            var entity = _repository.GetById(id);
            entity.IsDeleted = true;
            _uow.SaveChanges();
            return id;
        }

        protected readonly IUow _uow;
        protected readonly IRepository<{{ entity }}> _repository;
        protected readonly ICache _cache;
    }
}
